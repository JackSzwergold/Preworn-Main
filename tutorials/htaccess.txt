AddDefaultCharset UTF-8

Options +FollowSymlinks
RewriteEngine on
RewriteCond %{REQUEST_FILENAME} !-d
RewriteCond %{REQUEST_FILENAME} !-f

# This redirects any call to this subdirectory to an index file.
# The '?controller=$1' passes the pathname as an HTML param called 'id'.
RewriteRule     ^([a-z0-9-]+)/?$ index.php?controller=$1 [L]
